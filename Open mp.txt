#include "fractal_openmp.h"
#include "palette.h"

#include <omp.h>
#include <complex>
#include <immintrin.h>
#include <cmath>

extern int max_iteraciones;

//--------------------------------------------------------
// Función de acotado para Mandelbrot (Escalar)
//--------------------------------------------------------
uint32_t acotado_mandelbrot_openmp(double cr, double ci) {
    /*
      Ecuación de Mandelbrot: z0 = 0, zn+1 = zn^2 + c
      donde c = cr + i*ci es la posición del píxel en el plano complejo
    */
    int iter = 1;
    double zr = 0.0;
    double zi = 0.0;

    while (iter < max_iteraciones && (zr * zr + zi * zi) < 4.0) {
        double dr = zr * zr - zi * zi + cr;
        double di = 2.0 * zr * zi + ci;

        zr = dr;
        zi = di;

        iter++;
    }

    if (iter < max_iteraciones) {
        int index = iter % PALETTE_SIZE; // Índice en la paleta de colores
        return color_ramp[index];
    }
    return 0xFF000000; // Negro para los puntos dentro del conjunto
}

//--------------------------------------------------------
// 1. Mandelbrot con división de regiones manuales en OpenMP
//--------------------------------------------------------
void mandelbrot_openmp_regiones(double x_min, double x_max, double y_min, double y_max, uint32_t width, uint32_t height, uint32_t* pixel_buffer) {
    double dx = (x_max - x_min) / width;
    double dy = (y_max - y_min) / height;

    #pragma omp parallel
    {
        int thread_id = omp_get_thread_num();
        int thread_count = omp_get_num_threads();

        int delta = std::ceil(width * 1.0 / thread_count);
        int start = thread_id * delta;
        int end = (thread_id + 1) * delta;

        if (thread_id == thread_count - 1) {
            end = width; // El último hilo absorbe las columnas sobrantes
        }

        for (int i = start; i < end; i++) {
            for (int j = 0; j < height; j++) {
                double x = x_min + i * dx;
                double y = y_max - j * dy;

                auto color = acotado_mandelbrot_openmp(x, y);
                pixel_buffer[j * width + i] = color;
            }
        }
    }
}

//--------------------------------------------------------
// 2. Mandelbrot con #pragma omp parallel for y schedule dinámico
//--------------------------------------------------------
void mandelbrot_openmp_for(double x_min, double x_max, double y_min, double y_max, uint32_t width, uint32_t height, uint32_t* pixel_buffer) {
    double dx = (x_max - x_min) / width;
    double dy = (y_max - y_min) / height;

    // Se usa schedule(dynamic, 16) para corregir el desbalance de carga del fractal
    #pragma omp parallel for schedule(dynamic, 16)
    for (int i = 0; i < static_cast<int>(width); i++) {
        for (int j = 0; j < static_cast<int>(height); j++) {
            double x = x_min + i * dx;
            double y = y_max - j * dy;

            auto color = acotado_mandelbrot_openmp(x, y);
            pixel_buffer[j * width + i] = color;
        }
    }
}

//--------------------------------------------------------
// 3. Mandelbrot con SIMD (AVX2) + OpenMP
//--------------------------------------------------------
void mandelbrot_openmp_for_simd(double x_min, double x_max, double y_min, double y_max, uint32_t width, uint32_t height, uint32_t* pixel_buffer) {
    float dx = static_cast<float>((x_max - x_min) / width);
    float dy = static_cast<float>((y_max - y_min) / height);

    __m256 xmin = _mm256_set1_ps(static_cast<float>(x_min));
    __m256 ymax = _mm256_set1_ps(static_cast<float>(y_max));

    __m256 sxcale = _mm256_set1_ps(dx);
    __m256 sycale = _mm256_set1_ps(dy);

    __m256 max_norma = _mm256_set1_ps(4.0f);
    __m256 one = _mm256_set1_ps(1.0f);
    __m256 two = _mm256_set1_ps(2.0f);

    #pragma omp parallel for schedule(dynamic)
    for (int j = 0; j < static_cast<int>(height); j++) {
        __m256 my = _mm256_set1_ps(static_cast<float>(j));
        __m256 ci = _mm256_sub_ps(ymax, _mm256_mul_ps(my, sycale));

        for (int i = 0; i < static_cast<int>(width); i += 8) {
            // Evaluamos 8 píxeles continuos en el eje X
            __m256 mx = _mm256_set_ps(
                static_cast<float>(i + 7), static_cast<float>(i + 6),
                static_cast<float>(i + 5), static_cast<float>(i + 4),
                static_cast<float>(i + 3), static_cast<float>(i + 2),
                static_cast<float>(i + 1), static_cast<float>(i + 0)
            );

            __m256 cr = _mm256_add_ps(xmin, _mm256_mul_ps(mx, sxcale));

            // Para Mandelbrot, z inicia en (0,0)
            __m256 zr = _mm256_setzero_ps();
            __m256 zi = _mm256_setzero_ps();

            int iter = 1;
            __m256 mk = _mm256_set1_ps(1.0f);

            while (iter < max_iteraciones) {
                __m256 zr2 = _mm256_mul_ps(zr, zr);
                __m256 zi2 = _mm256_mul_ps(zi, zi);
                __m256 zrzi = _mm256_mul_ps(zr, zi);

                // z_next = z^2 + c
                __m256 new_zr = _mm256_add_ps(_mm256_sub_ps(zr2, zi2), cr);
                __m256 new_zi = _mm256_add_ps(_mm256_mul_ps(two, zrzi), ci);

                zr = new_zr;
                zi = new_zi;

                __m256 zr2_n = _mm256_mul_ps(zr, zr);
                __m256 zi2_n = _mm256_mul_ps(zi, zi);
                __m256 norma2 = _mm256_add_ps(zr2_n, zi2_n);

                // Máscara |z|^2 <= 4.0
                __m256 mask = _mm256_cmp_ps(norma2, max_norma, _CMP_LE_OQ);

                mk = _mm256_add_ps(_mm256_and_ps(mask, one), mk);

                if (_mm256_movemask_ps(mask) == 0) {
                    break; // Todos los 8 puntos escaparon
                }

                iter++;
            }

            float d[8];
            _mm256_storeu_ps(d, mk);

            for (int it = 0; it < 8; it++) {
                int px = i + it;
                if (px < static_cast<int>(width)) {
                    int index = j * width + px;
                    if (d[it] < max_iteraciones) {
                        int color_index = static_cast<int>(d[it]) % PALETTE_SIZE;
                        pixel_buffer[index] = color_ramp[color_index];
                    } else {
                        pixel_buffer[index] = 0xFF000000;
                    }
                }
            }
        }
    }
}
